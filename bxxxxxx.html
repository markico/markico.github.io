[tex]\boxed{x = \frac{15}{4} }[/tex] opsi tidak ada

.

Langkah penyelesaian

[tex]\frac{4^{2x - 3}~.~2^{x - 3}}{4 .2^{2 - x}} = 4^{x + 1}[/tex]

[tex]\frac{(2^2)^{2x - 3}~.~2^{x - 3}}{(2^2) .2^{2 - x}} = (2^2)^{x + 1}[/tex]

[tex]\frac{2^{2(2x - 3)}~.~2^{x - 3}}{2^{2 - x + 2}} = 2^{2(x + 1)}[/tex]

[tex]\frac{2^{4x - 6}~.~2^{x - 3}}{2^{4 - x}} = 2^{2x + 2}[/tex]

[tex]\frac{2^{4x - 6 + (x - 3)}}{2^{4 - x}} = 2^{2x + 2}[/tex]

[tex]\frac{2^{5x - 9}}{2^{4 - x}} = 2^{2x + 2}[/tex]

[tex]2^{5x - 9 - (4 - x)} = 2^{2x + 2}[/tex]

[tex]2^{5x - 9 - 4 + x} = 2^{2x + 2}[/tex]

[tex]2^{6x - 13} = 2^{2x + 2}[/tex]

[tex]6x - 13 = 2x + 2[/tex]

[tex]6x - 2x = 2 + 13[/tex]

[tex]4x = 15[/tex]

[tex]x = \frac{15}{4}[/tex]
