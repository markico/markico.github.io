v0 = 80 Km/jm

v0 = 800/36 m/s

v = 0 (berhenti)

t = 5 menit

t = 300 s

[tex]a = \frac{v - v0}{t} \\ a = \frac{0 - \frac{800}{36} }{300} \\ a = - \frac{800}{36} \times 300 \\ a = - 6666.7 \frac{m}{ {s}^{2} } [/tex]
