6. Dik : C = 100 + 0,8Yd
             I = 100
            G = 200
            X = 400
            T = 0,25 Y
           M = 0,1 Y
           Kondisi full employment Y = 1800
a.  Yd = Y - T
          = Y - 0,25Y
         = 0,75 Y
                                 C = 100 + 0,8 (0,75Y)
                                 C = 100 + 0,6 Y
b.  AE = C + I + G + X - M
           = 100 + 0,6Y + 100 + 200 + 400 - 0,1Y
           = 800 + 0,5Y
                                  Y = AE
                                  Y = 800 + 0,5Y
                                 0,5Y = 800
                                   Y = 1600
c.  1.  C = 100 + 0,8 (Y - (0,25Y + T)
          C = 100 + 0,8 (0,75 Y - T)
          C = 100 + 0,6Y - 0,8T
          AE = C + I + G + X - M
                = 800 + 0,5 Y - 0,8T
                                           Y = AE
                                           Y = 800 + 0,5Y - T
                                          0,5 Y = 800 - T
                                         0,5 (1800) = 800 - 0,8T 
                                              900 = 800 - 0,8T
                                             100 = -0,8T
                                              -125 = T
         Maka diperlukan pengurangan pajak sebesar 125
  2.  G = 200 + G1
       AE = 800 + 0,5Y + G1
                                           Y = AE
                                          Y = 800 + 0,5Y + G1
                                         0,5Y = 800 + G1
                                        0,5 (1800) = 800 + G1
                                        900 = 800 + G1
       Maka, pemerintah akan menambah pengeluaran pemerintah sebesar 100

d.  Kedudukan budget awal ekuilibrium= T - G
                                                              = 0,25Y - 200
                                                             = 0,25 (1600) - 200
                                                              = 400 - 200
                                                             = 200 (Surplus)
     Kedudukan budget pada full employment = T - G
                                                                       = 0,25 Y - (200 + 100)
                                                                      = 0,25 (1800) - 300
                                                                     = 450 - 300
                                                                     = 150 (Surplus)
     Kedudukan budget jika menggunakan kondisi I = T - G
                                                                           = -125 + 0,25Y - 200
                                                                           = 0,25 (1800) - 325
                                                                           = 450 - 325
                                                                           = 125 (Surplus)
     Fungsi pajak baru = -125 + 0,25Y
e.  Posisi neraca perdagangan = X - M
                                                 = 400 - 0,1Y
                                                = 400 - 0,1 (1600)
                                               = 400 - 160
                                               = 240 (Surplus)
f. Nilai multiplier = ΔY / ΔG = 200 / 100 = 2
