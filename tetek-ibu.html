1 - 2 + 3 - 4 + 5 - 6 + 7 - 8 + ... - 100

= 1 + [ (- 2 + 3) + (- 4 + 5) + (- 6 + 7) + (- 8 + 9) + ... + (-98 + 99) ] - 100

= 1 + [ 1 + 1 + 1 + 1 + .... + 1 ] - 100

= 1 + [ 49 x 1 ] - 100

= 1 + 49 - 100

= 50 - 100

= -50
