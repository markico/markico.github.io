Jawaban:

[tex]p + q + r = 160000000 \\ q - 100000 = p + 100000 \\ q = p + 200000 \\ r + 200000 = p + q \\ r + 200000 = p + (p + 200000) \\ r + 200000 = 2p + 200000 \\ r = 2p[/tex]

p+q+r = 160000000

p + (p+200000) +2p = 160000000

4p = 160000000 - 200000

4p = 159800000

p= 159800000 ÷ 4

p= 39950000

r= 2p

r= 2 ×39950000

r= 7990000
