kecepatan = 60 km/jam
berangkat = 08.15
dit; tiba/waktu = ?
jawab;
waktu = jarak: kecepatan
waktu = 30 km: 60 km / jam
waktu = 1/2 jam
1/2 jam = 30 menit = 00.30

tiba = berangkat + lama perjalanan
tiba = 08.15+00.30
= 08.45
